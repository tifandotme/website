## Generating Lorem Ipsum

Here's a simple TypeScript function to generate Lorem Ipsum `console.log(foo` text:

```ts
type GeneratorOptions = {
  paragraphs: number
  wordsPerParagraph: number
  startWithLoremIpsum?: boolean
}

function generateLoremIpsum(options: GeneratorOptions): string {
  const words = [
    "lorem",
    "ipsum",
    "dolor",
    "sit",
    "amet",
    "consectetur",
    "adipiscing",
    "elit",
    "sed",
    "do",
    "eiusmod",
    "tempor",
    "incididunt",
  ]

  let result = ""

  for (let i = 0; i < options.paragraphs; i++) {
    let paragraph = ""

    for (let j = 0; j < options.wordsPerParagraph; j++) {
      const randomIndex = Math.floor(Math.random() * words.length)
      const word = words[randomIndex]

      if (j === 0) {
        paragraph += word.charAt(0).toUpperCase() + word.slice(1)
      } else {
        paragraph += " " + word
      }
    }

    result += paragraph + "\n\n"
  }

  return result.trim()
}
```
